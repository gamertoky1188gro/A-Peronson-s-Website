  117 | async function collectCodeFiles(dirPath, collector, limit = MAX_FILES_TO_SCAN) {
  118 |   if (collector.length >= limit) return
  119 | 
  120 |   const entries = await fs.readdir(dirPath, { withFileTypes: true })
  121 |   for (const entry of entries) {
  122 |     if (collector.length >= limit) return
  123 | 
  124 |     const resolvedPath = path.join(dirPath, entry.name)
  125 |     if (entry.isDirectory()) {
  126 |       if (SKIP_DIRECTORIES.has(entry.name)) continue
  127 |       await collectCodeFiles(resolvedPath, collector, limit)
  128 |       continue
  129 |     }
  130 | 
  131 |     if (!entry.isFile()) continue
  132 |     const extension = path.extname(entry.name).toLowerCase()
  133 |     if (!CODE_EXTENSIONS.has(extension)) continue
  134 |     collector.push(resolvedPath)
  135 |   }

  153 | function findBestSnippet(content, tokens) {
  154 |   const lines = content.split('\n')
  155 |   let best = { score: 0, line: '', lineNumber: 1 }
  156 | 
  157 |   for (let index = 0; index < lines.length; index += 1) {
  158 |     const rawLine = lines[index]
  159 |     const normalizedLine = normalize(rawLine)
  160 |     if (!normalizedLine) continue
  161 | 
  162 |     let score = 0
  163 |     for (const token of tokens) {
  164 |       if (normalizedLine.includes(token)) score += 1
  165 |     }
  166 | 
  167 |     if (score > best.score) {
  168 |       best = {
  169 |         score,
  170 |         line: sanitizeString(rawLine.trim(), MAX_SNIPPET_LENGTH),
  171 |         lineNumber: index + 1,
  172 |       }
  173 |     }
  174 |   }
  175 | 
  176 |   return best.score > 0 ? best : null
  177 | }

  179 | async function searchCodeContext(questionText) {
  180 |   const tokens = tokenize(questionText)
  181 |   if (tokens.length === 0) {
  182 |     return {
  183 |       summary: '',
  184 |       snippets: [],
  185 |       prompt_context: '',
  186 |     }
  187 |   }
  188 | 
  189 |   const files = await getCodeFiles()
  190 |   const matches = []
  191 | 
  192 |   for (const filePath of files) {
  193 |     let stat
  194 |     try {
  195 |       stat = await fs.stat(filePath)
  196 |     } catch {
  197 |       continue
  198 |     }
  199 |     if (!stat.isFile() || stat.size > MAX_FILE_BYTES) continue
  200 | 
  201 |     let content = ''
  202 |     try {
  203 |       content = await fs.readFile(filePath, 'utf8')
  204 |     } catch {
  205 |       continue
  206 |     }
  207 | 
  208 |     const relativePath = path.relative(process.cwd(), filePath)
  209 |     const fileTokens = tokenize(relativePath)
  210 | 
  211 |     let score = 0
  212 |     for (const token of tokens) {
  213 |       if (fileTokens.includes(token)) score += 2
  214 |       if (normalize(content).includes(token)) score += 1
  215 |     }
  216 | 
  217 |     if (score === 0) continue
  218 | 
  219 |     const bestSnippet = findBestSnippet(content, tokens)
  220 |     matches.push({
  221 |       path: relativePath,
  222 |       score: score + (bestSnippet?.score || 0),
  223 |       snippet: bestSnippet,
  224 |     })
  225 |   }
  226 | 
  227 |   const topMatches = matches
  228 |     .sort((a, b) => b.score - a.score)
  229 |     .slice(0, MAX_MATCHED_SNIPPETS)
  230 | 
  231 |   const snippets = topMatches.map((match) => ({
  232 |     file: match.path,
  233 |     score: match.score,
  234 |     line: match.snippet?.lineNumber || null,
  235 |     snippet: match.snippet?.line || '',
  236 |   }))
  237 | 
  238 |   const summary = snippets
  239 |     .map((item) => `${item.file}${item.line ? `:${item.line}` : ''}`)
  240 |     .join(', ')
  241 | 
  242 |   let promptContext = snippets
  243 |     .map((item) => `[${item.file}${item.line ? `:${item.line}` : ''}] ${item.snippet}`)
  244 |     .join('\n')
  245 |   promptContext = sanitizeString(promptContext, MAX_CONTEXT_CHARS)
  246 | 
  247 |   return {
  248 |     summary,
  249 |     snippets,
  250 |     prompt_context: promptContext,
  251 |   }
  252 | }

  254 | function findBestKeywordRule(questionText, rules = []) {
  255 |   const normalizedQuestion = normalize(questionText)
  256 |   const compactQuestion = compactTokenKey(normalizedQuestion)
  257 |   const rawQuestionTokens = normalizedQuestion.split(/[^a-z0-9]+/).filter(Boolean)
  258 |   const compactQuestionTokens = new Set(rawQuestionTokens.map((token) => compactTokenKey(token)).filter(Boolean))
  259 | 
  260 |   let bestRule = null
  261 |   let bestRuleScore = 0
  262 | 
  263 |   for (const rule of rules) {
  264 |     const normalizedKeywords = rule.keywords.map((keyword) => compactTokenKey(keyword)).filter(Boolean)
  265 |     let score = 0
  266 |     for (const keyword of normalizedKeywords) {
  267 |       if (!keyword) continue
  268 |       if (compactQuestionTokens.has(keyword)) {
  269 |         score += 1
  270 |         continue
  271 |       }
  272 |       if (keyword.length > 3 && compactQuestion.includes(keyword)) score += 1
  273 |     }
  274 | 
  275 |     if (score > bestRuleScore) {
  276 |       bestRule = rule
  277 |       bestRuleScore = score
  278 |     }
  279 |   }
  280 | 
  281 |   return { bestRule, bestRuleScore }
  282 | }
