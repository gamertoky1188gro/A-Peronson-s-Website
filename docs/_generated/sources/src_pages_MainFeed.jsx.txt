   37 | function formatRelativeTime(value) {
   38 |   // Convert an ISO timestamp into a short "Just now / 5m ago / 2h ago" label for feed cards.
   39 |   if (!value) return ''
   40 |   const date = new Date(value)
   41 |   if (Number.isNaN(date.getTime())) return ''
   42 |   const diffMs = Date.now() - date.getTime()
   43 |   const diffMinutes = Math.floor(diffMs / 60000)
   44 |   if (diffMinutes < 1) return 'Just now'
   45 |   if (diffMinutes < 60) return `${diffMinutes}m ago`
   46 |   const diffHours = Math.floor(diffMinutes / 60)
   47 |   if (diffHours < 24) return `${diffHours}h ago`
   48 |   const diffDays = Math.floor(diffHours / 24)
   49 |   return `${diffDays}d ago`
   50 | }

   52 | function normalizeFeedItem(raw) {
   53 |   // Backend feed rows can be buyer requests or company products.

   56 |   const isBuyerRequest = entityType === 'buyer_request'

   58 |   const accountType = raw.company_role || (isBuyerRequest ? 'buyer' : 'factory')

   63 |     entityType,

   66 |       name: raw.author?.name || raw.company_name || raw.organization_name || raw.org || raw.name || 'Unknown',

   87 | async function copyToClipboard(text) {
   88 |   // Utility used for "Copy link" / "Copy details" actions.
   89 |   // Uses modern Clipboard API when available, with a DOM fallback for older browsers.
   90 |   if (!text) return false
   91 |   if (navigator.clipboard?.writeText) {
   92 |     await navigator.clipboard.writeText(text)
   93 |     return true
   94 |   }
   95 | 
   96 |   const el = document.createElement('textarea')
   97 |   el.value = text
   98 |   el.setAttribute('readonly', 'true')
   99 |   el.style.position = 'fixed'
  100 |   el.style.left = '-9999px'
  101 |   document.body.appendChild(el)
  102 |   el.select()
  103 |   const ok = document.execCommand('copy')
  104 |   document.body.removeChild(el)
  105 |   return ok
  106 | }
