   45 | export async function recordView(userId, productId, { windowMinutes = 10 } = {}) {
   46 |   const viewerId = sanitizeString(String(userId || ''), 120)
   47 |   const pid = sanitizeString(String(productId || ''), 120)
   48 |   if (!viewerId || !pid) return 'not_found'
   49 | 
   50 |   const products = await readJson(PRODUCTS_FILE)
   51 |   const product = safeArray(products).find((p) => String(p.id) === pid) || null
   52 |   if (!product) return 'not_found'
   53 | 
   54 |   const all = safeArray(await readJson(FILE))
   55 |   const now = Date.now()
   56 |   const windowMs = Math.max(1, Number(windowMinutes) || 10) * 60 * 1000
   57 | 
   58 |   const existingIndex = all.findIndex((row) => row.user_id === viewerId && row.product_id === pid)
   59 |   if (existingIndex >= 0) {
   60 |     const lastAt = new Date(all[existingIndex].viewed_at || 0).getTime()
   61 |     if (Number.isFinite(lastAt) && (now - lastAt) < windowMs) {
   62 |       return { ok: true, deduped: true, viewed_at: all[existingIndex].viewed_at }
   63 |     }
   64 |     all[existingIndex].viewed_at = toIsoNow()
   65 |     all[existingIndex].updated_at = toIsoNow()
   66 |   } else {
   67 |     all.push({
   68 |       id: crypto.randomUUID(),
   69 |       user_id: viewerId,
   70 |       product_id: pid,
   71 |       viewed_at: toIsoNow(),
   72 |       created_at: toIsoNow(),
   73 |       updated_at: toIsoNow(),
   74 |     })
   75 |   }
   76 | 
   77 |   await writeJson(FILE, all)
   78 |   await trackEvent({ type: 'product_viewed', actor_id: viewerId, entity_id: pid })
   79 |   return { ok: true, deduped: false }
   80 | }

   82 | export async function listMyProductViews(userId, { cursor = 0, limit = 10 } = {}) {
   83 |   const viewerId = sanitizeString(String(userId || ''), 120)
   84 |   const safeCursor = Math.max(0, Math.floor(Number(cursor || 0)))
   85 |   const safeLimit = Math.min(50, Math.max(1, Math.floor(Number(limit || 10))))
   86 | 
   87 |   const [views, products, users] = await Promise.all([
   88 |     readJson(FILE),
   89 |     readJson(PRODUCTS_FILE),
   90 |     readJson(USERS_FILE),
   91 |   ])
   92 | 
   93 |   const viewsForUser = safeArray(views).filter((v) => v.user_id === viewerId).sort(sortNewest)
   94 |   const productsById = new Map(safeArray(products).map((p) => [String(p.id), normalizeProductVideoFlags(p)]))
   95 |   const usersById = new Map(safeArray(users).map((u) => [String(u.id), u]))
   96 | 
   97 |   const pageRows = viewsForUser.slice(safeCursor, safeCursor + safeLimit)
   98 |   const items = pageRows.map((v) => {
   99 |     const product = productsById.get(String(v.product_id)) || null
  100 |     const author = product ? publicAuthor(usersById.get(String(product.company_id))) : null
  101 |     return {
  102 |       id: v.id,
  103 |       viewed_at: v.viewed_at,
  104 |       product: product ? {
  105 |         id: product.id,
  106 |         title: product.title,
  107 |         category: product.category,
  108 |         material: product.material,
  109 |         moq: product.moq,
  110 |         lead_time_days: product.lead_time_days,
  111 |         description: product.description,
  112 |         hasVideo: Boolean(product.hasVideo),
  113 |         video_url: product.video_url || '',
  114 |         video_review_status: product.video_review_status || '',
  115 |       } : null,
  116 |       author,
  117 |     }
  118 |   }).filter((row) => row.product)
  119 | 
  120 |   const nextCursor = safeCursor + safeLimit < viewsForUser.length ? safeCursor + safeLimit : null
  121 |   return {
  122 |     cursor: safeCursor,
  123 |     next_cursor: nextCursor,
  124 |     total: viewsForUser.length,
  125 |     items,
  126 |   }
  127 | }
