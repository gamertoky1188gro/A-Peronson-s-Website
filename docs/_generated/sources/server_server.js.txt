  101 | function registerSocketUser(socket, userId) {
  102 |   if (!userId) return

  107 | function unregisterSocketUser(socket, userId) {

  191 | async function joinChatRoom(socket, payload) {
  192 |   const matchId = String(payload?.match_id || '').trim()
  193 |   const user = parseSocketUser(payload?.token)
  194 | 
  195 |   if (!matchId) {
  196 |     sendWs(socket, { type: 'chat_error', error: 'match_id is required to join chat room' })
  197 |     return
  198 |   }
  199 | 
  200 |   if (!user?.id) {
  201 |     sendWs(socket, { type: 'chat_error', error: 'Valid token is required to join chat room' })
  202 |     return
  203 |   }
  204 | 
  205 |   const canJoin = await canAccessMatch(matchId, user.id)
  206 |   if (!canJoin) {
  207 |     sendWs(socket, { type: 'chat_error', error: 'Forbidden: thread access denied' })
  208 |     return
  209 |   }
  210 | 
  211 |   leaveChatRoom(socket)
  212 | 
  213 |   if (!chatRooms.has(matchId)) {
  214 |     chatRooms.set(matchId, new Set())
  215 |   }
  216 | 
  217 |   socket.userId = user.id
  218 |   registerSocketUser(socket, user.id)
  219 |   setUserOnline(user.id)
  220 |   const canSend = await canAccessMatch(matchId, socket.userId)
  221 |   if (!canSend) {
  222 |     sendWs(socket, { type: 'chat_error', error: 'Forbidden: thread access denied' })
  223 |     return
  224 |   }
  225 | 
  226 |   const room = chatRooms.get(matchId)
  227 |   const participants = [...room].map((participantSocket) => participantSocket.userId).filter(Boolean)
  228 |   room.add(socket)
  229 |   socket.chatRoomId = matchId
  230 |   touchUser(user.id)

  233 |   sendWs(socket, {
  234 |     type: 'joined_chat_room',
  235 |     match_id: matchId,
  236 |     participant_id: user.id,
  237 |     participants,
  238 |     messages: history,
  239 |   })

  241 |   for (const peer of room) {
  242 |     if (peer === socket) continue
  243 |     sendWs(peer, {
  244 |       type: 'chat_participant_joined',

  285 | async function joinCallRoom(socket, payload) {
  286 |   const callId = String(payload?.call_id || '').trim()
  287 |   const tokenUser = parseSocketUser(payload?.token)
  288 |   const participantId = String(payload?.participant_id || '').trim() || tokenUser?.id || `anon-${Date.now()}`
  289 |   if (!callId) {
  290 |     sendWs(socket, { type: 'call_error', error: 'call_id is required to join room' })
  291 |     return
  292 |   }
  293 | 
  294 |   if (!tokenUser?.id) {
  295 |     sendWs(socket, { type: 'call_error', error: 'Valid token is required to join call room' })
  296 |     return
  297 |   }
  298 | 
  299 |   const call = await getCallSession(callId, tokenUser.id)
  300 |   if (!call || call === 'forbidden') {
  301 |     sendWs(socket, { type: 'call_error', error: 'Forbidden: call access denied' })
  302 |     return
  303 |   }
  304 | 
  305 |   leaveCallRoom(socket)
  306 | 
  307 |   if (!callRooms.has(callId)) {
  308 |     callRooms.set(callId, new Set())
  309 |   }
  310 | 
  311 |   const room = callRooms.get(callId)
  312 |   const existingParticipants = [...room].map((s) => s.participantId).filter(Boolean)
  313 |   room.add(socket)
  314 |   socket.callRoomId = callId
  315 |   socket.participantId = participantId
  316 |   socket.userId = tokenUser.id
  317 |   registerSocketUser(socket, tokenUser.id)
  318 | 
  319 |   sendWs(socket, {
  320 |     type: 'joined_call_room',
  321 |     call_id: callId,
  322 |     participant_id: participantId,
  323 |     participants: existingParticipants,
  324 |     should_offer: existingParticipants.length > 0,
  325 |   })
  326 | 
  327 |   for (const peer of room) {
  328 |     if (peer === socket) continue
  329 |     sendWs(peer, {
  330 |       type: 'participant_joined',
  331 |       call_id: callId,
  332 |       participant_id: participantId,
  333 |     })
  334 |   }
