   28 |     const row = {
   29 |       request_id: requestId,
   30 |       locked_by: agent.id,
   31 |       allowed_agents: [agent.id],
   32 |       created_at: new Date().toISOString(),
   33 |     }
   34 |     all.push(row)
   35 |     await writeJson(FILE, all)
   36 |     await createLockNotification(agent.id, `You claimed buyer request ${requestId}.`, requestId, agent.id)
   37 |     return { status: 'claimed', ...row }
   38 |   }
   39 | 
   40 |   if (existing.locked_by === agent.id || existing.allowed_agents.includes(agent.id)) {
   41 |     return { status: 'granted', ...existing }
   42 |   }
   43 | 
   44 |   return { status: 'locked', notification: 'Conversation is locked by another agent.' }
   45 | }
